			{

				{
					"ModifierName"	"modifier_phased"
					"Target"		"TARGET"
					"Duration"		"0.03"
				}

				"RunScript"
				{
					"ScriptFile"    "buildings/rally_point.lua"
					"Function"      "MoveToRallyPoint"
				}

				"RunScript"
				{
					"ScriptFile"    "units/human_upgrades.lua"
					"Function"      "ApplyForgedSwordsUpgrade"
				}
				"RunScript"
				{
					"ScriptFile"    "units/human_upgrades.lua"

				}
			}
